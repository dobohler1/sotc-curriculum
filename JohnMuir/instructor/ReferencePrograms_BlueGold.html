Program BLUE — Drive Forward

from hub import light_matrix
import motor_pair
import runloop

async def main():
    motor_pair.pair(motor_pair.PAIR_1, 0, 1)
    light_matrix.write("GO")
    await motor_pair.move_for_time(motor_pair.PAIR_1, 2000, 0, velocity=50)
    light_matrix.write("DONE")

runloop.run(main())

Program GOLD — Beep & Spin

from hub import light_matrix, sound
import motor_pair
import runloop

async def main():
    motor_pair.pair(motor_pair.PAIR_1, 0, 1)
    await sound.beep(800, 500, 100)
    await motor_pair.move_for_time(motor_pair.PAIR_1, 1500, 100, velocity=30)

runloop.run(main())
